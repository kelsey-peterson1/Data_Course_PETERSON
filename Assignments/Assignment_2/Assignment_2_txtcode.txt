#find all files in Data that begin with lowercase 'b' 

b_files <- list.files(path = "Data", pattern =  "^b", full.names = TRUE, recursive = TRUE)

#Display the first line of each 'b' file using for-loop

for (file in b_files) {
  first_line_b <- readLines(file, n=1)
  cat("File:", file, "\n")
  cat("First line:", first_line_b, "\n\n")
  
}

# display first lines of all .csv files using for loop

  first_line_all <- readLines(file, n=1)

}
